<script>

let locationValue = "";


function startFlow() {

  document.querySelector('.main-btn').style.display = "none";

  document.getElementById('step1').style.display = "block";

}


function next(step) {

  document.querySelectorAll('.step').forEach(s => s.style.display = "none");

  document.getElementById('step' + step).style.display = "block";

}


document.getElementById('reason').addEventListener('change', function() {

  document.getElementById('customReason').style.display =

    this.value === 'custom' ? 'block' : 'none';

});


function getLocation() {

  navigator.geolocation.getCurrentPosition(pos => {

    locationValue = pos.coords.latitude + "," + pos.coords.longitude;

    document.getElementById("locationText").innerText = locationValue;

  });

}


function submitData() {

  const amount = document.getElementById('amount').value;


  let reason = document.getElementById('reason').value;

  if (reason === 'custom') {

    reason = document.getElementById('customReason').value;

  }


  const paymentMode = document.getElementById('paymentMode').value;

  const payer = document.getElementById('payer').value;


  const beneficiaries = Array.from(

    document.getElementById('beneficiers').selectedOptions

  ).map(o => o.value);


  const payload = {

    amount,

    reason,

    location: locationValue,

    paymentMode,

    payer,

    beneficiaries: beneficiaries.join(",")

  };


  fetch("PASTE_YOUR_SCRIPT_URL_HERE", {

    method: "POST",

    body: JSON.stringify(payload)

  })

  .then(() => {

    alert("Saved ✅");

    location.reload();

  });

}
